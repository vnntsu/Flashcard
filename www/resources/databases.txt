CREATE TABLE vocabulary(
	idvocab integer PRIMARY KEY,
	text varchar(50),
	pronounce text,
	image text,
	sound text,
	isremember int,
	rememberday date,
	iscustom int
);

CREATE TABLE pronunciation(
	idpronun INTEGER PRIMARY KEY,
	text varchar(5),
	image text,
	sound text, 
	type int, 
	idvocab int
);

CREATE TABLE topic(
	idtopic INTEGER PRIMARY KEY AUTOINCREMENT,
	name varchar(50),
	describe varchar(255),
	image text
);

CREATE TABLE subtopic(
	idsubtopic int primary key autoincrement,
	idtopic int,
	name text
);

CREATE TABLE topicofvocab(
	idvocab int,
	idsubtopic int,
	constraint pk_topicofword primary key(idvocab, idsubtopic)
);

CREATE TABLE level(
	idlevel INTEGER PRIMARY KEY AUTOINCREMENT,
	point int,
	nameoflevel int,
	image varchar(255)
);

CREATE TABLE profile(
	firstname varchar(100) primary key,
	lastname varchar(100),
	wordperday int,
	avatar text,
	idlevel int,
	currentexp int
);

CREATE TABLE kindofword(
	idkindofword integer primary key autoincrement,
	name varchar(50),
	vnname varchar(50)
);

CREATE TABLE typeofword(
	idvocab int,
	idkindword int,
	example varchar(255),
	meaning varchar(255),
	vnmean varchar(255),
	constraint pk_typeID primary key (idvocab,idkindword)
);

CREATE TABLE achievement(
	idachieve INTEGER PRIMARY KEY AUTOINCREMENT,
	name varchar(50),
	objective varchar(255),
	describe varchar(255),
	image varchar(255),
	isachieve int
);
